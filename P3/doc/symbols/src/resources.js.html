 10 /**
 11  * @class Resources This is a simple an image loading utility.
 12  * @name Resources
 13  * @description This is a simple an image loading utility. It eases the process of loading
 14  * image files so that they can be used within your game. It also includes
 15  * a simple "caching" layer so it will reuse cached images if you attempt
 16  * to load the same image multiple times.
 17  */
 18 (function() {
 19     /**
 20      * @lends Resources
 21      */
 22      /**
 23       * @var resourceCache
 24       * @description A dictionary matching urls with accociated items
 25       * of type Image
 26       */
 27     var resourceCache = {};
 28 
 29      /**
 30       * @var audioCache
 31       * @description A dictionary matching urls with accociated items
 32       * of type Audio
 33       */
 34     var audioCache = {};
 35 
 36     var loading = [];
 37 
 38     /**
 39       * @var readyCallbacks
 40       * @description An array of functions to be called when the calling window
 41       * calls its onReady method.
 42       * of type Image
 43       */
 44     var readyCallbacks = [];
 45 
 46     /**
 47      * @function load
 48      * @description This is the publicly accessible image loading function. It accepts
 49      * either an array of strings pointing to image files or a string for a single
 50      * image. It will then call our private image loading function accordingly.
 51      * @param {string or string[]} urlOrArr Either a string or an array of strings representing
 52      * url of files to cache.
 53      */
 54     function load(urlOrArr) {
 55         if(urlOrArr instanceof Array) {
 56             /* If the developer passed in an array of images
 57              * loop through each value and call our image
 58              * loader on that image file
 59              */
 60             urlOrArr.forEach(function(url) {
 61                 _load(url);
 62             });
 63         } else {
 64             /* The developer did not pass an array to this function,
 65              * assume the value is a string and call our image loader
 66              * directly.
 67              */
 68             _load(urlOrArr);
 69         }
 70     }
 71 
 72     /**
 73      * @function loadaudio
 74      * @description This is the publicly accessible audio loading function. It accepts
 75      * either an array of strings pointing to audio files or a string for a single
 76      * audio. It will then call our private audio loading function accordingly.
 77      * @param {string or string[]} urlOrArr Either a string or an array of strings representing
 78      * url of files to cache.
 79      */
 80     function loadaudio(urlOrArr) {
 81         if(urlOrArr instanceof Array) {
 82             /* If the developer passed in an array of images
 83              * loop through each value and call our image
 84              * loader on that image file
 85              */
 86             urlOrArr.forEach(function(url) {
 87                 _loadaudio(url);
 88             });
 89         } else {
 90             /* The developer did not pass an array to this function,
 91              * assume the value is a string and call our image loader
 92              * directly.
 93              */
 94             _loadaudio(urlOrArr);
 95         }
 96     }
 97 
 98     /**
 99      * @function _load
100      * @description This is our private image loader function, it is
101      * called by the public image loader function. If this URL has been
102      * previously loaded it will exist within
103      * our resourceCache array. Just return that image rather
104      * re-loading the image.
105      * @param {string or string[]} url Either a single url or an array of urls.
106      * @see resourceCache
107      */
108     function _load(url) {
109         if(resourceCache[url]) {
110             /* If this URL has been previously loaded it will exist within
111              * our resourceCache array. Just return that image rather
112              * re-loading the image.
113              */
114             return resourceCache[url];
115         } else {
116             /* This URL has not been previously loaded and is not present
117              * within our cache; we'll need to load this image.
118              */
119             var img = new Image();
120             img.onload = function() {
121                 /* Once our image has properly loaded, add it to our cache
122                  * so that we can simply return this image if the developer
123                  * attempts to load this file in the future.
124                  */
125                 resourceCache[url] = img;
126 
127                 /* Once the image is actually loaded and properly cached,
128                  * call all of the onReady() callbacks we have defined.
129                  */
130                 if(isEverythingReady()) {
131                     readyCallbacks.forEach(function(func) { func(); });
132                 }
133             };
134 
135             /* Set the initial cache value to false, this will change when
136              * the image's onload event handler is called. Finally, point
137              * the images src attribute to the passed in URL.
138              */
139             resourceCache[url] = false;
140             img.src = url;
141         }
142     }
143 
144     /**
145      * @function _loadaudio
146      * @description This is our private audio loader function, it is
147      * called by the public audio loader function. If this URL has been
148      * previously loaded it will exist within
149      * our audioCache array. Just return that image rather
150      * re-loading the image.
151      * @param {string or string[]} url Either a single url or an array of urls.
152      * @see audioCache
153      */
154     function _loadaudio(url) {
155         if(audioCache[url]) {
156             /* If this URL has been previously loaded it will exist within
157              * our resourceCache array. Just return that image rather
158              * re-loading the image.
159              */
160              console.log("audio already loaded");
161             return audioCache[url];
162         } else {
163             /* This URL has not been previously loaded and is not present
164              * within our cache; we'll need to load this image.
165              */
166             console.log("creating audio object");
167             var aud = new Audio(url);
168             aud.onloadeddata = function() {
169                 console.log("audio objects onload function was called");
170                 /* Once our image has properly loaded, add it to our cache
171                  * so that we can simply return this image if the developer
172                  * attempts to load this file in the future.
173                  */
174                 audioCache[url] = aud;
175 
176                 /* Once the image is actually loaded and properly cached,
177                  * call all of the onReady() callbacks we have defined.
178                  */
179                 if(isEverythingReady()) {
180                     readyCallbacks.forEach(function(func) { func(); });
181                 }
182             };
183 
184             /* Set the initial cache value to false, this will change when
185              * the image's onload event handler is called. Finally, point
186              * the images src attribute to the passed in URL.
187              */
188             audioCache[url] = false;
189             aud.src = url;
190         }
191     }
192 
193     /**
194      * @function get
195      * @description This is used by developer's to grab references to images they know
196      * have been previously loaded. If an image is cached, this functions
197      * the same as calling load() on that URL.
198      * @param {string} url URL used as a key to look up a cached image.
199      * @returns {Image} The requested Image object.
200      */
201     function get(url) {
202         return resourceCache[url];
203     }
204 
205     /**
206      * @function getaudio
207      * @description This is used by developer's to grab references to audio they know
208      * have been previously loaded. If an audio is cached, this functions
209      * the same as calling load() on that URL.
210      * @param {string} url URL used as a key to look up a cached audio.
211      * @returns {Image} The requested Audio object.
212      */
213     function getaudio(url) {
214         console.log("performing getaudio for " + url);
215         console.log("will return " + audioCache[url]);
216         return audioCache[url];
217     }
218 
219     /**
220      * @function isReady
221      * @description This function determines if all of the images that have been requested
222      * for loading have in fact been completed loading.
223      * @returns {bool} <b>true</b> if all requested images have been loaded, otherwise <b>false</b>.
224      */
225     function isReady() {
226         var ready = true;
227         for(var k in resourceCache) {
228             if(resourceCache.hasOwnProperty(k) &&
229                !resourceCache[k]) {
230                 ready = false;
231             }
232         }
233         return ready;
234     }
235 
236         /**
237      * @function isAudioReady
238      * @description This function determines if all of the audio that has been requested
239      * for loading have in fact been completed loading.
240      * @returns {bool} <b>true</b> if all requested audio has been loaded, otherwise <b>false</b>.
241      */
242     function isAudioReady() {
243         var ready = true;
244         for(var k in audioCache) {
245             if(audioCache.hasOwnProperty(k) &&
246                !audioCache[k]) {
247                 ready = false;
248             }
249         }
250         console.log("isAudioReady returning " + ready);
251         return ready;
252     }
253 
254     function isEverythingReady() {
255         return isReady() && isAudioReady();
256     }
257 
258     /**
259      * @function onReady
260      * @description This function will add a function to the callback stack that is called
261      * when all requested images are properly loaded.
262      * @param {function} func A function to add to the readyCallbacks stack.
263      */
264     function onReady(func) {
265         readyCallbacks.push(func);
266     }
267 
268     /* This object defines the publicly accessible functions available to
269      * developers by creating a global Resources object.
270      */
271     window.Resources = {
272         load: load,
273         loadaudio: loadaudio,
274         get: get,
275         getaudio: getaudio,
276         onReady: onReady,
277         isReady: isEverythingReady
278     };
279 })();
