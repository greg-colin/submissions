 12 /**
 13  * @class GameRectangle Used for character positioning within game tiles.
 14  * @description Character bounding boxes within game tiles. Everything within the game
 15  * is set within a transparent tile which is larger than the object. These rectangles
 16  * are used to hold the bounding boxes for the actual game object.
 17  * @param {int} x - The left bound
 18  * @param {int} y - The top bound
 19  * @param {int} w - The width
 20  * @param {int} h - The height
 21  */
 22 var GameRectangle = function(x, y, w, h) {
 23     /** 
 24      * @member {int}
 25      * @description The X offset between the character's tile and the bounding rectangle within it.
 26      */
 27     this.x = x;
 28     /** 
 29      * @member {int}
 30      * @description The Y offset between the character's tile and the bounding rectangle within it.
 31      */
 32     this.y = y;
 33     /** 
 34      * @member {int}
 35      * @description The width of the character's bounding rectangle.
 36      */
 37     this.w = w;
 38     /** 
 39      * @member {int}
 40      * @description The height of the character's bounding rectangle.
 41      */
 42     this.h = h;
 43 }

126 /**
127  * @method hasCollidedWith
128  * @type Bool
129  * @description Determines if the current character has collided with another,
130  * given as a parameter. The bounding boxes within the character tiles are
131  * checked to see if they overlap.
132  * @param {GameCharacter} objcharacter
133  * @returns {Bool} indicating whether of not the character this has collided with objcharacter
134  */
135 GameCharacter.prototype.hasCollidedWith = function(objcharacter) {
136     // Precalculate everything, to make it easier to reason about the values and allow
137     // for later, more interesting collision detection
138     var top1 = this.boundingRect.y;
139     var bottom1 = top1 + this.boundingRect.h;
140     var left1 = this.boundingRect.x;
141     var right1 = left1 + this.boundingRect.w;
142     var top2 = objcharacter.boundingRect.y;
143     var bottom2 = top2 + objcharacter.boundingRect.h;
144     var left2 = objcharacter.boundingRect.x;
145     var right2 = left2 + objcharacter.boundingRect.w; 
146 
147     if (left1 < right2 && right1 > left2 && top1 < bottom2 && bottom1 > top2) {
148             return true;
149     }
150     return false;    
151 }

252 /**
253  * @method update
254  * @description Update enemy parameters for each iteration of the Engine's main loop.
255  * Movement is multiplied by the floating point parameter dt (set in the engine), to
256  * ensure that the game operates at the same speed for all players. Characters do not
257  * move when the global variable showBoundingRects is set to true. Values are updated
258  * for both the character tile and the character bounding box within it. When the enemy
259  * moves offscreen to the right, its row is reset randomly and then it re-enters on
260  * the left side.
261  * <p>
262  * Enemy characters do not move if they are bumping the rear end of another enemy.
263  * @param {float} dt - Movement multiplier for speed equalization.
264  * @see isBumpingAnyOther
265  */
266 Enemy.prototype.update = function(dt) {
267     if (!showBoundingRects) {
268         if (!this.isBumpingAnyOther()) {
269             this.setLocation(this.x + this.speed * dt, this.y);
270         }
271         if (this.isOffscreenX(this.x)) {
272             this.x = 0;
273             this.row = randRange(1, 3);
274             this.y = this.row * boxHeight - enemyDrawYOffset;
275             this.speed = randRange(enemySpeedLow, enemySpeedHigh);
276         }
277         this.boundingRect.x = this.x + enemyBoundLeftOffset;
278         this.boundingRect.y = this.y + enemyBoundTopOffset;
279     }
280 }

282 /**
283  * @method isOffscreenX
284  * @description Determine if an enemy is offscreen in the X dimension. An enemy is offscreen
285  * when it is centered on the rightmost tile.
286  * @param {int} x The X position of the character tile.
287  * @returns {bool} True if the X position of the enemy tile places is off the canvas. Otherwise false.
288  */
289 Enemy.prototype.isOffscreenX = function(x) {
290     if ((x < 0) || (x > boxWidth * (boardWidthBoxes -1))) {
291         return true;
292     }
293     return false;
294 }

296 /**
297  * @method isBumpingBehind
298  * @description Returns true if the current enemy has collided with another (enemy can only
299  * collide in the X axis).
300  * @param {Enemy} otherenemy An enemy with which we are testing collision.
301  * @returns {bool} True if collision occured, otherwise false.
302  */
303 Enemy.prototype.isBumpingBehind = function(otherenemy) {
304     if (this.hasCollidedWith(otherenemy) && (this.boundingRect.x < otherenemy.boundingRect.x)) {
305         return true;
306     }
307     return false;
308 }
309 
310 /**
311  * @method isBumpingAnyOther
312  * @description Returns true if the current enemy has collided with any other enemy.
313  * @returns {bool} True if a collision has occurred, otherwise false.
314  * @see isBumpingBehind
315  */
316 Enemy.prototype.isBumpingAnyOther = function() {
317     for (var i = 0; i < numEnemy; i++) {
318         if (this.name != allEnemies[i].name) {
319             if (this.isBumpingBehind(allEnemies[i])) {
320                 return true;
321             }
322         }
323     }
324     return false;
325 }

358 /**
359  * @method update
360  * @description Update player parameters for each iteration of the Engine's main loop.
361  * Movement is multiplied by the floating point parameter dt (set in the engine), to
362  * ensure that the game operates at the same speed for all players. Characters do not
363  * move when the global variable showBoundingRects is set to true. Values are updated
364  * for both the character tile and the character bounding box within it. <b>NOTE: Player
365  * only moves in increments of box size (NOT tile size) in this game and those moves
366  * are made in handleInput(). Only the character bounding boxes are updated in this
367  * method.</b><p>
368  * If the player attempts to go offscreen in the up direction, <b>and</b> has collected
369  * all of the prizes on the current level, activity is frozen to prevent play while
370  * changing levels, and a countdown is set for timer ticks until the level changes.
371  * @see countdown
372  * @see frozen
373  * @see pauseBackgroundAudio
374  * @param {float} dt - Movement multiplier for speed equalization
375  */
376 Player.prototype.update = function(dt) {
377     this.boundingRect.x = this.x + playerBoundLeftOffset;
378     this.boundingRect.y = this.y + playerBoundTopOffset;
379     if (this.y <= 0 && !frozen) {
380         if (this.hasAllPrizes()) {
381             frozen = true;
382             pauseBackgroundAudio();
383             playNewLevelSound();
384             countdown = 5;
385         }
386     }
387 }

399 /**
400  * @method handleInput
401  * @description Keypresses accepted by the application are handled to
402  * move the player up, down, left, or right. Tests for and prohibits
403  * out-of-bounds moves.
404  * @param {string} inkey - String corresponding to the keycode for the key pressed.
405  */
406 Player.prototype.handleInput = function(inkey) {
407     console.log("valid keypress");
408     var snd = Resources.getaudio(audioUrls[1]);
409     snd.mediaGroup = "game";
410     if (!frozen) {
411         switch (inkey) {
412             case 'left':
413                 if (this.x > 0) {
414                     this.x = this.x - 101;
415                     snd.currentTime = 0;
416                     snd.play();
417                 }
418                 break;
419             case 'up':
420                 if (this.y > 0) {
421                     this.y = this.y - boxHeight;
422                     snd.currentTime = 0;
423                     snd.play();
424                 }
425             break;
426             case 'right':
427                 if (this.x < 404) {
428                     this.x = this.x + 101;
429                     snd.currentTime = 0;
430                     snd.play();
431                 }
432                 break;
433             case 'down':
434                 if (this.y < 404) {
435                     this.y = this.y + boxHeight;
436                     snd.currentTime = 0;
437                     snd.play();
438                 }
439                 break;
440             case 'a' :
441                 toggleAudio();
442                 break;
443             case 'b' :
444                 if (showBoundingRects) {
445                     showBoundingRects = false;
446                 } else {
447                     showBoundingRects = true;
448                 }
449                 break;
450             case 'c' :
451                 this.switchCharacter();
452                 break;
453             default:
454                 console.log("Key pressed: " + inkey);
455             break;
456         }
457     } else {
458         console.log("key while frozen");
459     }
460 }

477 /**
478  * @function tickTimer
479  * @description This timer gets called approximately every timerInterval seconds. Along with
480  * countdown (which is only counting down at a level completion) to create a pause between
481  * levels and count levels upward. Also tests for winning the game, and sets frozen (to
482  * disable player character movement) and status accordingly.
483  * @see timerInterval
484  * @see countdown
485  * @see level
486  */
487 function tickTimer() {
488     console.log("timer tick. countdown = " + countdown + " - frozen = " + frozen);
489     if (countdown != 0) {
490         countdown = countdown - 1;
491         if (countdown == 0) {
492             if (audioon) {
493                 startBackgroundAudio();
494             }
495             var _visibleprizes = 0;
496             for (var i = 0; i < numPrizes; i++) {
497                 if (allPrizes[i].isVisible) {
498                     _visiblePrizes = _visible_prizes + 1;
499                 }
500             }
501             if (level == 0) {
502                 level = level + 1;
503             } else {
504                 if (_visibleprizes == 0) {
505                     level = level + 1;
506                     if (level == 4) {
507                         frozen = true;
508                         status = statusEnum.won;
509                     }
510                 }
511             }
512             if (status == statusEnum.playing) {
513                 console.log("********** Calling for game reset");
514                 resetGame();
515             }
516         }
517     }
518 }

520 /**
521  * @function updateTimer
522  * @param {float} dx The interval, in seconds, of the time it took for the
523  * engine's main loop to perform the most recent requestAnimationFrame. 
524  * @description timerInterval (a value in seconds) is ticked down by the
525  * given parameter, and calls tickTimer every ~timerInterval seconds.
526  * value is used to tick down timerInterval.
527  * @see Engine
528  * @see tickTimer
529  * @see timerInterval
530  */
531 function updateTimer(dx) {
532     timer = timer - dx;
533     if (timer <= 0) {
534         tickTimer();
535         timer = timerInterval;
536     }
537 }

539 /**
540  * @function randRange
541  * @description Generate a random integer within a range given by the parameters. Used
542  * to generate a new row for enemy characters after they've move offscreen.
543  * @see Enemy#update
544  * @param {int} min - The lowest possible value
545  * @param {int} max - The highest possible value

548     return Math.floor(Math.random() * (max-min + 1) + min);

551 /**
552  * @function resetGame
553  * @description Dispose player and all enemies, and recreate them, restarting the game.
554  */
555 function resetGame() {
556     console.log("******************************** game reset");
557     if (level == 0) {
558         numEnemy = 1;
559         numPrizes = 0;
560     } else if (level == 1) {
561         numEnemy = 1;
562         numPrizes = 1;
563     } else {
564         numEnemy = level;
565         numPrizes = level;
566     }
567     allEnemies = [];
568     allPrizes = [];
569     player = null;
570     for (var i = 0; i < numEnemy; i++) {
571         var enemy = new Enemy("enemy" + i);
572         allEnemies.push(enemy);
573     }
574     for (var j = 0; j < numPrizes; j++) {
575         var prize = new Prize("prize" + j, randRange(1, 3), randRange(0, 4), j);
576         allPrizes.push(prize);
577     }
578     player = new Player("player");
579     frozen = false;
580     countdown = 0;
581 }

583 /**
584  * @function collisionDetectReact
585  * @description Determine if any enemy has collided with the player, and reset the game if so.
586  */
587 function collisionDetectReact() {
588     for (var i = 0; i < numEnemy; i++) {
589         if (allEnemies[i].hasCollidedWith(player)) {
590             playCollisionSound();
591             lives = lives - 1;
592             if (lives != 0) {
593                 resetGame();
594             } else {
595                 frozen = true;
596                 status = statusEnum.lost;
597             }
598         }
599     }
600     for (var j = 0; j < numPrizes; j++) {
601         if (allPrizes[j].isVisible) {
602             if (allPrizes[j].hasCollidedWith(player)) {
603                 playGetPrizeSound();
604                 score = score + allPrizes[j].value;
605                 allPrizes[j].isVisible = false;
606             }
607         }
608     }
609 }
