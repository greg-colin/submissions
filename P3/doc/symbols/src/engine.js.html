1 /**
  2  * @file engine.js
  3  * @fileoverview This file provides the game loop functionality (update entities and render),
  4  * draws the initial game board on the screen, and then calls the update and
  5  * render methods on your player and enemy objects (defined in your app.js).
  6  * <P>
  7  * A game engine works by drawing the entire game screen over and over, kind of
  8  * like a flipbook you may have created as a kid. When your player moves across
  9  * the screen, it may look like just that image/character is moving or being
 10  * drawn but that is not the case. What's really happening is the entire "scene"
 11  * is being drawn over and over, presenting the illusion of animation.
 12  * <P>
 13  * This engine is available globally via the Engine variable and it also makes
 14  * the canvas' context (ctx) object globally available to make writing app.js
 15  * a little simpler to work with.
 16  * @author Udacity, Inc.
 17  */

 19 /**
 20  * @class Provides the game loop functionality.
 21  * @description The Engine class provides the game loop functionality (update entities and render),
 22  * draws the initial game board on the screen, and then calls the update and
 23  * render methods on your player and enemy objects (defined in your app.js).
 24  */

 26     /* Predefine the variables we'll be using within this scope,
 27      * create the canvas element, grab the 2D context for that canvas
 28      * set the canvas elements height/width and add it to the DOM.

 62     /**
 63      * @var lastTime
 64      * @description The most recent time that the global Engine object

 66      * update() methods
 67      * @see Enemy#update
 68      * @see Player#update

 70     var lastTime;

 77     /**
 78      * @function main
 79      * @description The main loop of the game engine. This function serves as
 80      * the kickoff point for the game loop itself and handles properly calling
 81      * the update and render methods.
 82      * <p>
 83      * The loop sets a variable to the length of time between calls to
 84      * win.requestAnimation frame. This value is later passed to each
 85      * GameCharacter object's update method.
 86      * @see Player#update
 87      * @see Enemy#update
 88      */
 89     function main() {
 90         /* Get our time delta information which is required if your game
 91          * requires smooth animation. Because everyone's computer processes
 92          * instructions at different speeds we need a constant value that
 93          * would be the same for everyone (regardless of how fast their
 94          * computer is) - hurray time!
 95          */
 96         /**
 97          * @instance now
 98          * @description Get our time delta information which is required if your game
 99          * requires smooth animation. Because everyone's computer processes
100          * instructions at different speeds we need a constant value that
101          * would be the same for everyone (regardless of how fast their
102          * computer is) - hurray time!
103          */
104         var now = Date.now();
105 
106         var dt = (now - lastTime) / 1000.0;
107 
108         /* Call our update/render functions, pass along the time delta to
109          * our update function since it may be used for smooth animation.
110          */
111         update(dt);
112         render();
113 
114         /* Set our lastTime variable which is used to determine the time delta
115          * for the next time this function is called.
116          */
117         lastTime = now;
118 
119         /* Use the browser's requestAnimationFrame function to call this
120          * function again as soon as the browser is able to draw another frame.
121          */
122         win.requestAnimationFrame(main);
123     };
124

126      * @function init

139     /**
140      * @function update
141      * @param {float} dt window.requestAnimationFrame interval on the current machine, in seconds.
142      * @description This function is called by main (our game loop) and itself calls all
143      * of the functions which may need to update entity's data. Based on how
144      * you implement your collision detection (when two entities occupy the
145      * same space, for instance when your character should die), you may find
146      * the need to add an additional function call here. For now, we've left
147      * it commented out - you may or may not want to implement this
148      * functionality this way (you could just implement collision detection
149      * on the entities themselves within your app.js file).
150      * <p>
151      * Collision detection is done right after all the entities update.
152      * @see collisionDetectReact
153      */
154     function update(dt) {
155         updateEntities(dt);
156         updateTimer(dt);
157         collisionDetectReact();
158     }

160     /**
161      * @function updateEntities
162      * @param {float} dt window.requestAnimationFrame interval on the current machine, in seconds.
163      * @description This is called by the update function  and loops through all of the
164      * objects within your allEnemies array as defined in app.js and calls
165      * their update() methods. It will then call the update function for your
166      * player object. These update methods should focus purely on updating
167      * the data/properties related to  the object. Do your drawing in your
168      * render methods.
169      */
170     function updateEntities(dt) {
171         allEnemies.forEach(function(enemy) {
172             enemy.update(dt);
173         });
174         player.update();
175     }

177     /**
178      * @function render
179      * @description This function initially draws the "game level", it will then call
180      * the renderEntities function. Remember, this function is called every
181      * game tick (or loop of the game engine) because that's how games work -
182      * they are flipbooks creating the illusion of animation but in reality
183      * they are just drawing the entire screen over and over.
184      */
185     function render() {
186         /* This array holds the relative URL to the image used
187          * for that particular row of the game level.
188          */
189         var rowImages = [
190                 'images/water-block.png',   // Top row is water
191                 'images/stone-block.png',   // Row 1 of 3 of stone
192                 'images/stone-block.png',   // Row 2 of 3 of stone
193                 'images/stone-block.png',   // Row 3 of 3 of stone
194                 'images/grass-block.png',   // Row 1 of 2 of grass
195                 'images/grass-block.png'    // Row 2 of 2 of grass
196             ],
197             numRows = 6,
198             numCols = 5,
199             row, col;
200 
201         /* Loop through the number of rows and columns we've defined above
202          * and, using the rowImages array, draw the correct image for that
203          * portion of the "grid"
204          */
205         for (row = 0; row < numRows; row++) {
206             for (col = 0; col < numCols; col++) {
207                 /* The drawImage function of the canvas' context element
208                  * requires 3 parameters: the image to draw, the x coordinate
209                  * to start drawing and the y coordinate to start drawing.
210                  * We're using our Resources helpers to refer to our images
211                  * so that we get the benefits of caching these images, since
212                  * we're using them over and over.
213                  */
214                 ctx.drawImage(Resources.get(rowImages[row]), col * 101, row * 83);
215             }
216         }
217         renderEntities();
218         if (status == statusEnum.playing) {
219             renderScore();
220         }
221     }
222 
223     /**
224      * @function renderEntities
225      * @description This function is called by the render function and is called on each game
226      * tick. It's purpose is to redraw the background and score, then call the render
227      * functions you have defined on your character entities within app.js
228      * @see GameCharacter#render
229      * @see Enemy#render
230      */
231     function renderEntities() {
232         /* Loop through all of the objects within the allEnemies array and call
233          * the render function you have defined.
234          */
235         if (status == statusEnum.playing) {
236             allPrizes.forEach(function(prize) {
237                 prize.render();
238             });
239 
240             allEnemies.forEach(function(enemy) {
241                 enemy.render();
242             });
243 
244             player.render();
245         } else if (status == statusEnum.won) {
246             renderWinPoster();
247         } else {
248             renderLosePoster();
249         }
250     }
